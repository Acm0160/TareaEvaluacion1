package DII_T1;

import javax.swing.*;
import java.awt.*;

public class TareaEvaluacion1_Clasica extends JFrame {

    public TareaEvaluacion1_Clasica() {
        setTitle("Login Clásico");
        setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
        setSize(450, 350);
        setLocationRelativeTo(null);

        JPanel mainPanel = new JPanel(new GridBagLayout());
        mainPanel.setBackground(new Color(230, 240, 255));
        mainPanel.setBorder(BorderFactory.createEmptyBorder(20, 20, 20, 20));
        GridBagConstraints gbc = new GridBagConstraints();
        gbc.insets = new Insets(10, 10, 10, 10);
        gbc.fill = GridBagConstraints.HORIZONTAL;

        // Logo con fuente especial
        gbc.gridx = 0;
        gbc.gridy = 0;
        gbc.gridwidth = 2;
        gbc.anchor = GridBagConstraints.CENTER;
        JLabel logoLabel = new JLabel("Davante", SwingConstants.CENTER);
        logoLabel.setFont(new Font("Serif", Font.BOLD, 36)); // Fuente Serif, negrita, tamaño 36
        logoLabel.setForeground(new Color(0, 50, 150));
        mainPanel.add(logoLabel, gbc);

        // Campos de usuario y contraseña
        gbc.gridwidth = 1;
        gbc.gridx = 0;
        gbc.gridy = 1;
        gbc.anchor = GridBagConstraints.EAST;
        mainPanel.add(new JLabel("Usuario:"), gbc);

        gbc.gridx = 1;
        gbc.gridy = 1;
        gbc.anchor = GridBagConstraints.WEST;
        JTextField userField = new JTextField(15);
        mainPanel.add(userField, gbc);

        gbc.gridx = 0;
        gbc.gridy = 2;
        gbc.anchor = GridBagConstraints.EAST;
        mainPanel.add(new JLabel("Contraseña:"), gbc);

        gbc.gridx = 1;
        gbc.gridy = 2;
        gbc.anchor = GridBagConstraints.WEST;
        JPasswordField passwordField = new JPasswordField(15);
        mainPanel.add(passwordField, gbc);

        // JCheckBox
        gbc.gridx = 0;
        gbc.gridy = 3;
        gbc.gridwidth = 2;
        gbc.anchor = GridBagConstraints.CENTER;
        JCheckBox termsCheckBox = new JCheckBox("Aceptar términos y condiciones");
        termsCheckBox.setOpaque(false);
        mainPanel.add(termsCheckBox, gbc);

        // Botón
        gbc.gridx = 0;
        gbc.gridy = 4;
        gbc.gridwidth = 2;
        JButton accessButton = new JButton("Acceder");
        mainPanel.add(accessButton, gbc);

        add(mainPanel);
    }

    public static void main(String[] args) {
        SwingUtilities.invokeLater(() -> new TareaEvaluacion1_Clasica().setVisible(true));
    }
}
